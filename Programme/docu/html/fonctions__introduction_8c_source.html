31 /* Création d'une sous-chaîne du texte jusqu'à la lettre actuelle */
32 char buffer[indice + 2];
33 strncpy(buffer, ligne, indice);
34
35 /* Ajout de la lettre actuelle */
36 buffer[indice] = ligne[indice];

38 /* Ajout d'un caractère '\0' pour terminer la chaîne */
39 buffer[indice + 1] = '\0';

41 /* Rendu du texte actuel sur la surface texte */
42 (*surface) = TTF_RenderUTF8_Blended((*police), buffer, couleurBlanche);
43
44 /* Création de la texture texture_texte depuis la surface texte */
45 (*texture_texte) = SDL_CreateTextureFromSurface((*renderer), (*surface));
46
47 /* Récupération des dimensions du texte */
48 int largeur_texte, hauteur_texte;
49 SDL_QueryTexture((*texture_texte), NULL, NULL, &largeur_texte, &hauteur_texte);
50
51 /* Positionnement du texte au centre */
52 rectangle_texte_introduction->x = largeur / 2 - largeur_texte / 2;
53 rectangle_texte_introduction->y = hauteur / 2 - hauteur_texte ;
54 rectangle_texte_introduction->w = largeur_texte;
55 rectangle_texte_introduction->h = hauteur_texte;
56
57 /* Affichage de la texture texture_texte */
58 SDL_RenderCopy((*renderer), (*texture_texte), NULL, rectangle_texte_introduction);
59
60 SDL_FreeSurface((*surface));
61 SDL_DestroyTexture((*texture_texte));
62
63 /* Copie la texture de l'image du passer */
64 rectangle_passer->x = largeur - largeur / 21- largeur / 53;
65 rectangle_passer->y = hauteur / 30;
66 rectangle_passer->w = largeur / 21;
67 rectangle_passer->h = hauteur / 12;
68

70 erreur("Copie de la texture");
71
72 /* Fond noir */
73 SDL_SetRenderDrawColor((*renderer), 0, 0, 0, 0);
74
75 /* Affiche le rendu */
76 SDL_RenderPresent((*renderer));
