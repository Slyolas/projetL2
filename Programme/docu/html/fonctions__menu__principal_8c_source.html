221

227
228 break;
229
230 /* Gestion de l'événement de la position de la souris sur les différents items */
231 case SDL_MOUSEMOTION:
232
233 if((event->motion.x >= itemsMenu[0].rectangle.x) && (event->motion.x <= (itemsMenu[0].rectangle.x + itemsMenu[0].rectangle.w)) &&
234 (event->motion.y >= itemsMenu[0].rectangle.y) && (event->motion.y <= (itemsMenu[0].rectangle.y + itemsMenu[0].rectangle.h)))
235 (*selection_menu) = 1;
236
237 else if((event->motion.x >= itemsMenu[1].rectangle.x) && (event->motion.x <= (itemsMenu[1].rectangle.x + itemsMenu[1].rectangle.w)) &&
238 (event->motion.y >= itemsMenu[1].rectangle.y) && (event->motion.y <= (itemsMenu[1].rectangle.y + itemsMenu[1].rectangle.h)))
239 (*selection_menu) = 2;
240
241 else if((event->motion.x >= itemsMenu[2].rectangle.x) && (event->motion.x <= (itemsMenu[2].rectangle.x + itemsMenu[2].rectangle.w)) &&
242 (event->motion.y >= itemsMenu[2].rectangle.y) && (event->motion.y <= (itemsMenu[2].rectangle.y + itemsMenu[2].rectangle.h)))
243 (*selection_menu) = 3;
244
245 else
246 (*selection_menu) = 0;
247
248 break;
249
250 /* Gestion de l'événement du clic sur les différents items */
251 case SDL_MOUSEBUTTONDOWN:
252
253 if(tailleMenu == 2) {
254
255 if(clic_case((*event), itemsMenu[0].rectangle))
256 (*page_active) = NOUVELLE_PARTIE;
257
258 else if(clic_case((*event), itemsMenu[1].rectangle))
259 (*page_active) = OPTIONS;
260 }
261
262 else {
263
264 if(clic_case((*event), itemsMenu[0].rectangle))
265 (*page_active) = CARTE;
266
267 else if(clic_case((*event), itemsMenu[1].rectangle))
268 (*page_active) = NOUVELLE_PARTIE;
269

271 (*page_active) = OPTIONS;
272 }
273
274 /* Option plein écran */
275
276 if(clic_plein_ecran((*event), rectangle_plein_ecran, plein_ecran, window))
277 redimensionnement_fenetre((*event), largeur, hauteur);
278
279 break;
280
281 /* Gestion de l'événement du code de triche */
282 case SDL_KEYDOWN:
283
284 if((event->key.keysym.sym == SDLK_g) && (code_de_triche[1]))
285 code_de_triche[0] = 1;
286
287 if(event->key.keysym.sym == SDLK_b)
288 code_de_triche[1] = 1;
289
290 if((event->key.keysym.sym == SDLK_n) && (code_de_triche[0]))
291 code_de_triche[2] = 1;
292
293 break;
294
295 /* Quitter le programme */
296 case SDL_QUIT:
297 (*programme_lance) = SDL_FALSE;
298 break;
299
300 default:
301 break;
302 }
